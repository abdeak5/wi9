⚙️ تهيئة قاعدة البيانات والـ SQL في Supabase

لتخزين إعدادات الإعلانات ومقالاتك المنشورة ديناميكياً، يرجى نسخ الكود التالي ووضعه في محرّر الاستعلامات **(SQL Editor)** في لوحة تحكم حسابك على **Supabase** لتأسيس الجداول وإعطاء الصلاحيات الفورية لزوار الموقع للقراءة والكتابة: 

-- 1. جدول إعدادات الإعلانات adsense_config
CREATE TABLE IF NOT EXISTS adsense_config (
    id bigint PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
    client_id text DEFAULT 'ca-pub-1234567890123456',
    slot_sidebar text DEFAULT '1111111111',
    slot_top text DEFAULT '2222222222',
    slot_middle text DEFAULT '2222222222',
    slot_bottom text DEFAULT '3333333333',
    slot_inner_footer text DEFAULT '4444444444',
    ads_enabled boolean DEFAULT true,
    created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- إدخال الصف الافتراضي
INSERT INTO adsense_config (id, client_id, slot_sidebar, slot_top, slot_middle, slot_bottom, slot_inner_footer, ads_enabled)
VALUES (1, 'ca-pub-1234567890123456', '1111111111', '2222222222', '2222222222', '3333333333', '4444444444', true)
ON CONFLICT (id) DO NOTHING;

-- تفعيل سياسات الأمان RLS
ALTER TABLE adsense_config ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow anon read" ON adsense_config FOR SELECT USING (true);
CREATE POLICY "Allow anon update" ON adsense_config FOR UPDATE USING (true);
CREATE POLICY "Allow anon insert" ON adsense_config FOR INSERT WITH CHECK (true);

-- 2. جدول المقالات والتدوينات posts
CREATE TABLE IF NOT EXISTS posts (
    id bigint PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
    title text NOT NULL,
    content text NOT NULL,
    categories text[] DEFAULT '{}'::text[],
    image text,
    created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- تفعيل سياسات الأمان RLS للمقالات
ALTER TABLE posts ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow anon read posts" ON posts FOR SELECT USING (true);
CREATE POLICY "Allow anon insert posts" ON posts FOR INSERT WITH CHECK (true);
CREATE POLICY "Allow anon update posts" ON posts FOR UPDATE USING (true);
CREATE POLICY "Allow anon delete posts" ON posts FOR DELETE USING (true);

-- 3. جدول رسائل اتصل بنا messages
CREATE TABLE IF NOT EXISTS messages (
    id bigint PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
    name text NOT NULL,
    email text NOT NULL,
    subject text,
    message text NOT NULL,
    created_at timestamp with time zone DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- تفعيل سياسات الأمان RLS للرسائل
ALTER TABLE messages ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Allow anon read messages" ON messages FOR SELECT USING (true);
CREATE POLICY "Allow anon insert messages" ON messages FOR INSERT WITH CHECK (true);
CREATE POLICY "Allow anon delete messages" ON messages FOR DELETE USING (true);
